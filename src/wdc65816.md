static int memop(Node p, int cost)
{
    TRACE("memop\n");

    assert(p);
    assert(generic(p->op) == ASGN);
    assert(p->kids[0]);
    assert(p->kids[1]);
    if (generic(p->kids[1]->kids[0]->op) == INDIR
    && sametree(p->kids[0], p->kids[1]->kids[0]->kids[0]))
        return cost;
    else
        return LBURG_MAX;
}

static void function(Symbol f, Symbol caller[], Symbol callee[], int n)
{

    int i;
    TRACE("function\n");
    
    print("%s proc\n", f->x.name);

    //assert(caller[i] == 0);
    offset = maxoffset = 0;
    for (i = 0; callee[i]; i++) {
        Symbol p = callee[i];
        Symbol q = caller[i];
        assert(q);
        p->x.offset = q->x.offset = offset;
        p->x.name = q->x.name = stringf("%d", p->x.offset);
        p->sclass = q->sclass = AUTO;
        offset += roundup(q->type->size, 2);
    }




    gencode(caller, callee);
    //framesize = roundup(maxoffset, 4);

    emitcode();
    print("\tendp\n");
}

static void progbeg(int argc, char *argv[])
{
    unsigned i;

    TRACE("progbeg\n");

    {
        union {
            char c;
            int i;
        } u;
        u.i = 0;
        u.c = 1;
        swap = ((int)(u.i == 1)) != IR->little_endian;
    }
    parseflags(argc, argv);


    // up to 32 temporary "registers" on the dp.
    for (i = 0; i < 32; ++i)
    {
        reg16[i] = mkreg("@r%d", i, 0, IREG);
    }
    reg16w = mkwildcard(reg16);

    tmask[IREG] = 0xffffffff;
    tmask[FREG] = 0;  
    vmask[IREG] = 0;
    vmask[FREG] = 0;
 
    print("\tstring ASIS\n");
}

static void defsymbol(Symbol p)
{
    TRACE("defsymbol\n");


    if (p->scope >= LOCAL && p->sclass == STATIC)
        p->x.name = stringf("%s$%d", p->name, genlabel(1));
        
    else if (p->generated)
        p->x.name = stringf("__%s", p->name);
        
    else if (p->scope == GLOBAL || p->sclass == EXTERN)
        p->x.name = stringf("%s", p->name);
        
    else
        p->x.name = p->name;    
    
}

static Symbol rmap(int opk)
{
    TRACE("rmap\n");
    switch (optype(opk))
    {
    case 'B':
    case 'P':
        return reg32w;
        break;

    case I:
    case U:
        switch(opsize(opk))
        {
        case 1:
        case 2:
            return reg16w;
        case 4:
            return reg32w;
        case 8:
            return reg64w;
        }
        break;

    case F:
        break;
    }

    return 0;
}
