static int is_builtin(Node p, const char *fx) {
	Node arg = p->kids[0];
	if (generic(arg->op) == ADDRG 
		&& arg->syms[0] 
		&& arg->syms[0]->name == fx) return 0;

	return LBURG_MAX;
}

# unsigned __builtin_popcount(unsigned)
# Kernighan (sic) / Wegner algorithm
reg: CALLU2(address) {
	; %0
	pla
	ldx #0
	sta %c ; scratch
	cmp #0
	beq @fini
@loop:
	inx
	dec
	eor %c
	sta %c
	bne @loop
	stx %c
@fini:
} is_builtin(a, iigs_popcount)
