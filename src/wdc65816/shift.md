reg: RSHU2(reg, const) ^{

    unsigned value = constValue(kids[1]);

    if (value > 16) {
        EMIT(_("stz %c"));
        return;
    }
    EMIT("\tlda %0\n");

    if (value >= 8) {

        EMIT(
            _("xba")
            _("and #$00ff")
        );
        value -= 8;
    }

    while (value) {
        EMIT(_("lsr"));
        value--;
    }
    EMIT(_("sta %c"));

} 10
